--system repair—
bootrec /? for all commands
bcdedit /? for all commands
--rebuild bcd--
bcdedit /export c:\bcdbackup
attrib c:\boot\bcd –h –r –s
ren c:\boot\bcd bcd.old
bootrec /rebuildbcd
